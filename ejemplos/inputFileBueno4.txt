program ciclos;

{ Declaramos las variables }
var
  lenght, i, j, x, y, cont: Integer;
  arreglo: Array[1..50] of Integer;
  arreglo_bidi: Array[1..10, 1..5] of Integer;
  s: String;
  bool: boolean;

Function f(): Integer;
Begin
  f := 3;
end;

Procedure f2(a, b: integer);
Begin
  writeln('Hola');
end;

begin
  { Iniciamos las variables }
  lenght := 50;
  x := 10;
  y := 5;
  cont := 1;
  bool := TRUE;

  { Primer ciclo For }

  { Llenamos el arreglo bidimensional }
  For i := x Downto 1 Do
  Begin
    writeln('Estoy en ', i);
    For j := y Downto 1 Do
      writeln('Estoy en ', j);
  End;

  { Reiniciamos el contador }
  cont := 1;

  { Llenamos el arreglo unidimensional }
  For i := 1 To x Do
    Begin
        writeln('Estoy en ', i);
    End;

    For j := 1 To y Do
    Begin
      writeln('Estoy en ', j);
      cont := (cont + 1);
    End;

  { Reiniciamos el contador }
  cont := 1;

  { Imprimimos el arreglo unidimensional }
  While (cont <> lenght + 1) Do
  Begin
    writeln('Entro aqui');
    cont := cont + 1; { Faltaba incrementar el contador }
  End;

  writeln('Gracias por usar Pascal!');
end.
